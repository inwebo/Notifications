My Notifications version : 15-05-2012

Readme TOP

Ajoute un container au DOM courant pour y placer une liste ordonnée. Cette
liste contiendra des messages destinés au client. Cela peut être le résultat
d'une requête ajax, de traitements PHP ou javascript. Vous pouvez également
activé le debug pour avoir les traitements effectués dans la console de debug
de firebug (ou équivalents supportant console.log() ).

Usage TOP

			$(document).ready(function() {

			    // Nouvel objet par défaut
			    // var mypluginDefaut = new $.myNotifications();			    
			    
			    // Nouvel objet avec paramètres, affichage perpetuel des notifications
			    // Par la suite toutes les méthodes publics sont accessibles
			    var myplugin = new $.myNotifications({'delay':-1});


				// Démonstration
				$('#addMsgError').click(
		
					function() {
							myplugin.msgError('Message d\'erreur.');
					}
		
				);
	
				$('#addMsgWarning').click(
		
					function() {
							myplugin.msgWarning('Message warning!');
					}
		
				);
	
				$('#addMsgInfo').click(
		
					function() {
							myplugin.msgInfo('Message à caractère informatif!');
					}
		
				);
	
				$('#addMsgOkay').click(
		
					function() {
							myplugin.msgOkay('Tout est okay!');
					}
				);
	
				$('#addMsg').click(
		
					function() {
							myplugin.msg('Titre custom','Text custom');
					}
				);
	
			});
