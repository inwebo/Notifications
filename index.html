My Notifications

Error - Info - Warning - Okay - Message

Usage

$(document).ready(function() {

    // create a new instance of the plugin
    var myplugin = new $.myNotifications({'delay':-1});

	$('#addMsgError').click(
		
		function() {
				myplugin.msgError('Message d\'erreur.');
		}
		
	);
	
	$('#addMsgWarning').click(
		
		function() {
				myplugin.msgWarning('Message warning!');
		}
		
	);
	
	$('#addMsgInfo').click(
		
		function() {
				myplugin.msgInfo('Message à caractère informatif!');
		}
		
	);
	
	$('#addMsgOkay').click(
		
		function() {
				myplugin.msgOkay('Tout est okay!');
		}
	);
	
	$('#addMsg').click(
		
		function() {
				myplugin.msg('Titre custom','Text custom');
		}
	);
	
});
